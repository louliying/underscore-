JS源生isArray判断

        // js中 数组的判断
        var isArray = function (arr) {
            // 如果 支持Array.isArray, 则用源生Array.isArray()  这是ES5新增的
            // 如果不支持，则用toString返回值
            return Array.isArray? Array.isArray(arr) : Object.prototype.toString.call(arr) === '[object Array]';
        }
